# 1、点击按钮缩放动画

.activeBtn {
    &:active {
        animation: animateBtn 0.4s;
    }
}
@keyframes animateBtn {
    0% {
        transform: scale(1);
    }
    50% {
        transform: scale(0.86);
    }
    100% {
        transform: scale(1);
    }
}
