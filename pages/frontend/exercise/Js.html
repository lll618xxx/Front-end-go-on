# 程序执行结果判断题

# exapmle-1

var m = 1;
function add(n) {
  return (n = n + 1);
}
y = add(m);
function add(n) {
  return (n = n + 3);
}
z = add(m);
console.log(y, z);
// y和z的最终结果为: 4,4
// 解析：定义的函数会优先解析，而不是顺序解析
// y = add(m);语句和z = add(m); 语句调用的都是第二个add function

    console.log(this.s);
  }

  s: "teaher"
};

var getS = s.getS;
var getS1 = getS.bind(s);

s.getS(); // student
s.getS.apply(t); // teacher
getS(); // {s: "student", getS: ƒ} this指向window
