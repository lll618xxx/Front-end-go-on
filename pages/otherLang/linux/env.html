# 1、linux安装node

# 1-1、使用wget命令，下载文件

wget https://nodejs.org/dist/v14.16.1/node-v14.16.1-linux-x64.tar.xz
1

# 1-2、解压

tar -xf node-v14.16.1-linux-x64.tar.xz
1

# 1-3、配置环境变量

vim /etc/profile  如果没有权限，先加权限(sudo vim /etc/profile)
1
  • 添加如下语句
exexport NODE_HOME=/usr/local/nodejs // 这里是解压之后的node路径
export PATH=$NODE_HOME/bin:$PATH
1
2
  • 刷新权限
source /etc/profile
1

# 2、linux安装nginx

参考 (opens new window)

# 2-1、nginx操作命令

/usr/local/nginx/sbin
1

命令在进入上面的目录之后执行

  • 启动
./nginx
1
  • 修改配置后重启
./nginx -s reload
1
  • 关闭
ps -aux | grep 'nginx'
1
