# 1、简单的express后台API基础

/**
 * express 接口API设计基础
 */

const express = require('express')
const app = express()
const port = 8888;

// 配置body-parser后，req对象会多出一个属性body，通过req.body就可以获取表单的POST的数据
const bodyParser = require('body-parser')

// parse x-www-forn-urlencoded
app.use(bodyParser.urlencoded({ extended: false }))

 // parse json
app.use(bodyParser.json()) 

app.get('/', (req, res) => {
    res.send('sss')
})

// http://127.0.0.1:8888/see/43
app.get('/jsonapi/:id', (req, res) => {
    // { "id": "323"}
    res.send(req.params)
})

// http://127.0.0.1:8888/formapi?id=21&mes=cv
app.get('/formapi', (req, res) => {
    // {"id":"21","mes":"cv"}
    res.send(req.query)
})

app.post('/postapi', (req, res) => {
    res.send(req.body)
})

app.listen(port, () => {
    console.log(`项目运行在 ${port} 端口`)
})

# 2、Express路由容器

// index.js
/**
 * express 路由接口API设计提升
 */

const express = require('express')
const router = require('./router')
const app = express()
const port = 8888;

const bodyParser = require('body-parser')

app.use(bodyParser.urlencoded({ extended: false }))

app.use(bodyParser.json()) 

app.use(router)

app.listen(port, () => {
    console.log(`项目运行在 ${port} 端口`)
})
